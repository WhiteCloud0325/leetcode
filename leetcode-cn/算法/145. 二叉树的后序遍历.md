## 题目
给定一个二叉树，返回它的 后序 遍历。

示例:

输入: [1,null,2,3]  
   1
    \
     2
    /
   3 

输出: [3,2,1]
进阶: 递归算法很简单，你可以通过迭代算法完成吗？


## 答案

#### 方法一 递归
```
class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> res;
        helpFunc(root, res);
        return res;
    }

    void helpFunc(TreeNode *root, vector<int> &res) {
        if (root == NULL) return;

        helpFunc(root->left, res);
        helpFunc(root->right, res);
        res.push_back(root->val);
        return;
    }
};
```

#### 方法二 迭代 栈
```
class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> res;
        TreeNode* cur = root;
        stack<TreeNode*> st;
        while(!st.empty() || cur) {
            if (cur) {
                res.push_back(cur->val);
                if (cur->left) st.push(cur->left);
                cur= cur->right;
            }
            else {
                cur = st.top();
                st.pop();
            }
        }
        reverse(res.begin(), res.end());

        return res;
    }
};
```
